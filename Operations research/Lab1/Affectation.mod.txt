# Problem: Affectation avec prise en compte de préférences 

set Personnes;
set Taches;

# Constantes 
param note{i in Personnes, j in Taches};

# Variables 
var a{i in Personnes, j in Taches}, binary;

# Contraintes 
s.t. UniciteTachePersonne { i in Personnes }:
sum{j in Taches} a[i,j] = 1;

s.t. UnicitePersonneTache{ j in Taches }:
sum{i in Personnes} a[i,j] = 1;

# Objectif 
maximize Contentement :
    sum{j in Taches, i in Personnes} a[i,j]*note[i,j];

data;

# Définition  de la matrice de préférence 
set Personnes :=
P1
P2
P3;

set Taches := 
T1 
T2 
T3;

param note : T1 T2 T3 :=
P1 4 8 6
P2 7 1 3
P3 5 5 7;

end;
